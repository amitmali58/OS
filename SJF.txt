#include <stdio.h>

int main() {
    int n, i, time = 0, completed = 0;
    int at[10], bt[10], ct[10], wt[10], tat[10], done[10] = {0};
    float awt = 0, atat = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    for(i = 0; i < n; i++) {
        printf("AT & BT for P%d: ", i+1);
        scanf("%d%d", &at[i], &bt[i]);
    }

    while(completed < n) {
        int idx = -1, min = 9999;

        for(i = 0; i < n; i++)
            if(at[i] <= time && !done[i] && bt[i] < min)
                min = bt[i], idx = i;

        if(idx != -1) {
            time += bt[idx];           // run completely
            ct[idx] = time;
            tat[idx] = ct[idx] - at[idx];
            wt[idx] = tat[idx] - bt[idx];

            awt += wt[idx];
            atat += tat[idx];

            done[idx] = 1;
            completed++;
        } else time++;
    }

    printf("\nP\tAT\tBT\tCT\tTAT\tWT\n");
    for(i = 0; i < n; i++)
        printf("P%d\t%d\t%d\t%d\t%d\t%d\n",
               i+1, at[i], bt[i], ct[i], tat[i], wt[i]);

    printf("\nAvg WT=%.2f\nAvg TAT=%.2f\n", awt/n, atat/n);

    return 0;
}
