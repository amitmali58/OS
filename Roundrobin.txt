#include <stdio.h>
int main() {
    int n, i, time = 0, completed = 0, tq;
    int at[10], bt[10], rt[10];
    int ct[10], wt[10], tat[10];
    float awt = 0, atat = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    for(i = 0; i < n; i++) {
        printf("AT BT for P%d: ", i+1);
        scanf("%d%d", &at[i], &bt[i]);
        rt[i] = bt[i];
    }

    printf("Enter Time Quantum: ");
    scanf("%d", &tq);

    while(completed < n) {
        int executed = 0;

        for(i = 0; i < n; i++)
            if(at[i] <= time && rt[i] > 0) {
                executed = 1;

                if(rt[i] <= tq) {
                    time += rt[i];
                    ct[i] = time;
                    rt[i] = 0;
                    completed++;

                    tat[i] = ct[i] - at[i];
                    wt[i] = tat[i] - bt[i];
                    atat += tat[i];
                    awt += wt[i];
                } else {
                    time += tq;
                    rt[i] -= tq;
                }
            }

        if(!executed) time++;
    }

    printf("\nP\tAT\tBT\tCT\tTAT\tWT\n");
    for(i = 0; i < n; i++)
        printf("P%d\t%d\t%d\t%d\t%d\t%d\n",
               i+1, at[i], bt[i], ct[i], tat[i], wt[i]);

    printf("\nAvg WT=%.2f\nAvg TAT=%.2f\n", awt/n, atat/n);

    return 0;
}
