#include <stdio.h>

int main() {
    int n, i, time = 0, completed = 0;
    int at[10], bt[10], rt[10], pr[10];
    int ct[10], wt[10], tat[10];
    float awt = 0, atat = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    for(i = 0; i < n; i++) {
        printf("AT BT Priority for P%d: ", i+1);
        scanf("%d%d%d", &at[i], &bt[i], &pr[i]);
        rt[i] = bt[i];
    }

    while(completed < n) {
        int idx = -1, high = 9999;

        for(i = 0; i < n; i++)
            if(at[i] <= time && rt[i] > 0 && pr[i] < high)
                high = pr[i], idx = i;

        if(idx != -1) {
            rt[idx]--; time++;

            if(rt[idx] == 0) {
                completed++;
                ct[idx] = time;
                tat[idx] = ct[idx] - at[idx];
                wt[idx] = tat[idx] - bt[idx];
                atat += tat[idx];
                awt += wt[idx];
            }
        } else time++;
    }

    printf("\nP\tAT\tBT\tPR\tCT\tTAT\tWT\n");
    for(i = 0; i < n; i++)
        printf("P%d\t%d\t%d\t%d\t%d\t%d\t%d\n",
               i+1, at[i], bt[i], pr[i], ct[i], tat[i], wt[i]);

    printf("\nAvg WT=%.2f\nAvg TAT=%.2f\n", awt/n, atat/n);

    return 0;
}
