#include<stdio.h>
#include<stdlib.h>

int main() {
    int n, i, j, head, dir, total = 0;
    int req[50], temp;

    printf("Enter number of requests: ");
    scanf("%d", &n);

    printf("Enter request sequence: ");
    for(i = 0; i < n; i++) scanf("%d", &req[i]);

    printf("Enter current head: ");
    scanf("%d", &head);

    printf("Direction (1=right, 0=left): ");
    scanf("%d", &dir);

    // Sort requests
    for(i = 0; i < n-1; i++)
        for(j = 0; j < n-i-1; j++)
            if(req[j] > req[j+1]) {
                temp = req[j];
                req[j] = req[j+1];
                req[j+1] = temp;
            }

    printf("\nCurrent\tMove\n%d\t0\n", head);

    int start = head;

    // RIGHT direction (C-LOOK)
    if(dir == 1) {

        // Serve right side
        for(i = 0; i < n; i++)
            if(req[i] >= start) {
                printf("%d\t%d\n", req[i], abs(head - req[i]));
                total += abs(head - req[i]);
                head = req[i];
            }

        // Jump to lowest request (circular move, no cost added visually separate if needed)
        for(i = 0; i < n; i++)
            if(req[i] < start) {
                printf("%d\t%d\n", req[i], abs(head - req[i]));
                total += abs(head - req[i]);
                head = req[i];
            }
    } 

    // LEFT direction (C-LOOK)
    else {

        // Serve left side
        for(i = n-1; i >= 0; i--)
            if(req[i] <= start) {
                printf("%d\t%d\n", req[i], abs(head - req[i]));
                total += abs(head - req[i]);
                head = req[i];
            }

        // Jump to highest request
        for(i = n-1; i >= 0; i--)
            if(req[i] > start) {
                printf("%d\t%d\n", req[i], abs(head - req[i]));
                total += abs(head - req[i]);
                head = req[i];
            }
    }

    printf("\nTotal Seek Length = %d\n", total);
}
